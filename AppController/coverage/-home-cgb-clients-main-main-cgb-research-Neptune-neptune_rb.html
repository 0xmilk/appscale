548 def neptune_acquire_nodes(job_data)
 
549   # for jobs where no nodes need to be acquired (e.g., concurrent but not
 
550   # distributed programs), run them on the shadow node
 
551   if NO_NODES_NEEDED.include?(job_data["@type"])
 
552     Djinn.log_debug("No nodes needed for job type [#{job_data['@type']}]," +
 
553       " not acquiring nodes")
 
554     return [my_node]
 
555   end
 
556 
557   Djinn.log_debug("acquiring nodes")
 
558 
559   #num_of_vms_needed = optimal_nodes_hill_climbing(job_data, "performance")
 
560   nodes_needed = optimal_nodes(job_data)
 
561 
562   Djinn.log_debug("acquiring nodes for hybrid cloud neptune job")
 
563 
564   if nodes_needed.class == Array
 
565     nodes_needed = Hash[*nodes_needed]
 
566     Djinn.log_debug("request received to spawn hybrid nodes: #{nodes_needed.inspect}")
 
567   elsif nodes_needed.class == Fixnum
 
568     nodes_needed = {"cloud1" => nodes_needed}
 
569   else
 
570     Djinn.log_debug("nodes_needed was not the right class - should have been Array or Fixnum but was #{nodes_needed.class}")
 
571     # TODO: find a way to reject the job here
 
572   end
 
573 
574   nodes_to_use = []
 
575 
576   nodes_needed.each { |cloud, nodes_to_acquire|
 
577     # nodes_to_acquire can either be an integer or a URL
 
578     # if it's an integer, spawn up that many nodes
 
579     # if it's a URL, it refers to a remote cloud resource we don't control
 
580     # (e.g., Google App Engine), so skip it
 
581 
582     # in non-hybrid cloud runs, cloud1 will be the only cloud that specifies
 
583     # an integer value
 
584     if nodes_to_acquire =~ URL_REGEX
 
585       Djinn.log_debug("nodes to acquire for #{cloud} was a URL " + 
586         "[#{nodes_to_acquire}], so not spawning nodes")

591     nodes_for_cloud = neptune_find_open_nodes(cloud, nodes_to_acquire, job_data)
 
592     nodes_to_use = [nodes_to_use + nodes_for_cloud].flatten
 
593     # TODO: should check for failures acquiring nodes
 
594   }
 
595 
596   return nodes_to_use
 
597 end

738 def optimal_nodes(job_data)
 
739   return job_data["@nodes_to_use"]

741 
742 =begin
 
743 Hill Climbing Algorithm
 
744   # find minimum execution time t1
 
745   # find neighbors t0 and t2
 
746 
747   # if t0 is too low set it to t1
 
748   # if t2 is too high set it to t1
 
749 
750   # if no data for either, choose t2
 
751   # if no data for t0, choose t0
 
752   # if data for both, choose t1
 
753 =end
 
754 def optimal_nodes_hill_climbing(job_data, thing_to_optimize)
 
755   job_name = get_job_name(job_data)
 
756 
757   if thing_to_optimize != "cost" and thing_to_optimize != "performance"
 
758     abort("bad thing to optimize - can be cost or performance but was #{thing_to_optimize}")
 
759   end
 
760 
761   current_data = @neptune_jobs[job_name]
 
762   if current_data.nil? or current_data.empty?
 
763     Djinn.log_debug("neptune - no job data yet for [#{job_name}]")
 
764     return job_data["@nodes_to_use"]
 
765   end
 
766 
767   Djinn.log_debug("found job data for [#{job_name}]")
 
768 
769   min_val = INFINITY
 
770   optimal_job = nil
 
771   current_data.each { |job|
 
772     Djinn.log_debug("current job data is [#{job}]")
 
773 
774     if thing_to_optimize == "performance"
 
775       my_val = job.total_time
 
776     elsif thing_to_optimize == "cost"
 
777       my_val = job.cost
 
778     else
 
779       abort("bad thing to optimize again")
 
780     end
 
781 
782     if my_val < min_val
 
783       Djinn.log_debug("found a new minimum - [#{job}]")
 
784       optimal_job = job
 
785     end
 
786   }
 
787 
788   Djinn.log_debug("minimum is - [#{optimal_job}]")
 
789 
790   search_space = job_data["@can_run_on"]
 
791   t1 = optimal_job.num_nodes
 
792 
793   Djinn.log_debug("optimal right now is t1 = #{t1}")
 
794   t0, t2 = find_neighbors(t1, search_space)
 
795   Djinn.log_debug("t1's neighbors are #{t0} and t2 = #{t2}")
 
796 
797   d0 = get_job_data(job_name, t0)
 
798   d2 = get_job_data(job_name, t2)
 
799 
800   return t2 if d0.nil? and d2.nil?
 
801   return t0 if d0.nil?
 
802   return t1
 
803 end

805 def find_neighbors(val, search_space)
 
806   abort("no empty arrays") if search_space.nil? or search_space.empty?
 
807 
808   left, right = nil, nil
 
809   length = search_space.length
 
810   search_space.each_with_index { |item, index|
 
811     # set left
 
812     if index <= 0
 
813       left = val
 
814     else
 
815       left = search_space[index-1]
 
816     end
 
817 
818     # set right
 
819     if index < length - 1
 
820       right = search_space[index+1]
 
821     else
 
822       right = val
 
823     end
 
824 
825     break if item == val
 
826   }
 
827 
828   return left, right
 
829 end
