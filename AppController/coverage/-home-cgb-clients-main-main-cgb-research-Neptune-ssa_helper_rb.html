264 def variance(population)
 
265   n    = 0
 
266   mean = 0.0
 
267 
268   sum = population.reduce(0.0) do |sum, x|
 
269     n     += 1
 
270     delta  = x - mean
 
271     mean  += delta / n
 
272 
273     sum + delta * (x - mean)
 
274   end
 
275 
276   sum / n
 
277 end
