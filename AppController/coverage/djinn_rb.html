808   # Starts the given roles by using open nodes, spawning new nodes, or some
 
809   # combination of the two. 'nodes_needed' should be an Array, where each 
810   # item is an Array of the roles to start on each node.
 
811   def start_new_roles_on_nodes(nodes_needed, instance_type, secret)
 
812      if !valid_secret?(secret)
 
813       return BAD_SECRET_MSG
 
814     end
 
815 
816     if nodes_needed.class != Array
 
817       Djinn.log_debug("Was expecting nodes_needed to be an Array, not " +
 
818         "a #{nodes_needed.class}")
 
819       return BAD_INPUT_MSG
 
820     end
 
821 
822     Djinn.log_debug("Received a request to acquire nodes with roles " +
 
823       "#{nodes_needed.join(', ')}, with instance type #{instance_type} for " +
 
824       "new nodes")
 
825 
826     vms_to_use = []
 
827     ZKInterface.lock_and_run {
 
828       num_of_vms_needed = nodes_needed.length
 
829 
830       @nodes.each { |node|
 
831         if node.is_open?
 
832           Djinn.log_debug("Will use node #{node} to run new roles")
 
833           vms_to_use << node
 
834 
835           if vms_to_use.length == nodes_needed.length
 
836             Djinn.log_debug("Only using open nodes to run new roles")
 
837             break
 
838           end
 
839         end
 
840       }
 
841 
842       vms_to_spawn = nodes_needed.length - vms_to_use.length
 
843     
844       if vms_to_spawn > 0 and !is_cloud?
 
845         Djinn.log_debug("Still need #{vms_to_spawn} more nodes, but we " +
 
846         "aren't in a cloud environment, so we can't acquire more nodes - " +
 
847         "failing the caller's request.")
 
848         return NOT_ENOUGH_OPEN_NODES
 
849       end
 
850 
851       if vms_to_spawn > 0
 
852         Djinn.log_debug("Need to spawn up #{vms_to_spawn} VMs")
 
853         # start up vms_to_spawn vms as open
 
854         HelperFunctions.set_creds_in_env(@creds, "1")

856           @creds['machine'], instance_type, @creds['keyname'], 
857           @creds['infrastructure'], "cloud1", @creds['group'])
 
858

860         Djinn.log_debug("info about new nodes is " +
 
861           "[#{new_nodes_info.join(', ')}]")
 
862         add_nodes(new_nodes_info) 
863 
864         # add information about the VMs we spawned to our list, which may
 
865         # already have info about the open nodes we want to use
 
866         new_nodes = Djinn.convert_location_array_to_class(new_nodes_info,
 
867           @creds['keyname'])
 
868         vms_to_use << new_nodes
 
869         vms_to_use.flatten!
 
870       end
 
871     }
 
872 
873     wait_for_nodes_to_finish_loading(vms_to_use)
 
874     
875     nodes_needed.each_index { |i|
 
876       Djinn.log_debug("Adding roles #{nodes_needed[i].join(', ')} " +
 
877         "to virtual machine #{vms_to_use[i]}")
 
878       ZKInterface.add_roles_to_node(nodes_needed[i], vms_to_use[i])
 
879     }
 
880 
881     wait_for_nodes_to_finish_loading(vms_to_use)
 
882 
883     return "OK"
 
884   end
