837 =begin
 
838   Hill Climbing Algorithm
 
839     # find minimum execution time t1
 
840     # find neighbors t0 and t2
 
841 
842     # if t0 is too low set it to t1
 
843     # if t2 is too high set it to t1
 
844 
845     # if no data for either, choose t2
 
846     # if no data for t0, choose t0
 
847     # if data for both, choose t1
 
848 =end
 
849   def optimal_nodes_hill_climbing(job_data, thing_to_optimize)
 
850     job_name = get_job_name(job_data)
 
851 
852     if thing_to_optimize != "cost" and thing_to_optimize != "performance"
 
853       abort("bad thing to optimize - can be cost or performance but was #{thing_to_optimize}")
 
854     end
 
855 
856     current_data = @jobs[job_name]
 
857     if current_data.nil? or current_data.empty?
 
858       self.log("neptune - no job data yet for [#{job_name}]")
 
859       return job_data["@nodes_to_use"]
 
860     end
 
861 
862     self.log("found job data for [#{job_name}]")
 
863 
864     min_val = INFINITY
 
865     optimal_job = nil
 
866     current_data.each { |job|
 
867       self.log("current job data is [#{job}]")
 
868 
869       if thing_to_optimize == "performance"
 
870         my_val = job.total_time
 
871       elsif thing_to_optimize == "cost"
 
872         my_val = job.cost
 
873       else
 
874         abort("bad thing to optimize again")
 
875       end
 
876 
877       if my_val < min_val
 
878         self.log("found a new minimum - [#{job}]")
 
879         optimal_job = job
 
880       end
 
881     }
 
882 
883     self.log("minimum is - [#{optimal_job}]")
 
884 
885     search_space = job_data["@can_run_on"]
 
886     t1 = optimal_job.num_nodes
 
887 
888     self.log("optimal right now is t1 = #{t1}")
 
889     t0, t2 = find_neighbors(t1, search_space)
 
890     self.log("t1's neighbors are #{t0} and t2 = #{t2}")
 
891 
892     d0 = get_job_data(job_name, t0)
 
893     d2 = get_job_data(job_name, t2)
 
894 
895     return t2 if d0.nil? and d2.nil?
 
896     return t0 if d0.nil?
 
897     return t1
 
898   end

901   def find_neighbors(val, search_space)
 
902     abort("no empty arrays") if search_space.nil? or search_space.empty?
 
903 
904     left, right = nil, nil
 
905     length = search_space.length
 
906     search_space.each_with_index { |item, index|
 
907       # set left
 
908       if index <= 0
 
909         left = val
 
910       else
 
911         left = search_space[index-1]
 
912       end
 
913 
914       # set right
 
915       if index < length - 1
 
916         right = search_space[index+1]
 
917       else
 
918         right = val
 
919       end
 
920 
921       break if item == val
 
922     }
 
923 
924     return left, right
 
925   end
